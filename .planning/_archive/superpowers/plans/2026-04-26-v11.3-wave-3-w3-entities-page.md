```vue
<script setup lang="ts">
import { computed, watch } from 'vue';
import { useRoute, useRouter } from 'vue-router';
import { useHead } from '@unhead/vue';
import EntityBadge from '@/components/ui/EntityBadge.vue';
import GeneBadge from '@/components/ui/GeneBadge.vue';
import DiseaseBadge from '@/components/ui/DiseaseBadge.vue';
import SectionCard from '@/components/ui/SectionCard.vue';
import { useEntityRecord } from '@/composables/useEntityRecord';
import { useEntityStatus } from '@/composables/useEntityStatus';
import { useEntityReview } from '@/composables/useEntityReview';
import { useEntityPublications } from '@/composables/useEntityPublications';
import { useEntityPhenotypes } from '@/composables/useEntityPhenotypes';
import { useEntityVariation } from '@/composables/useEntityVariation';
import { useGeneRecord } from '@/composables/useGeneRecord';

const route = useRoute();
const router = useRouter();

const entityIdStr = computed(() => String(route.params.entity_id ?? ''));

// All six entity-side hooks fire on tick 0.
const entity = useEntityRecord(entityIdStr);
const status = useEntityStatus(entityIdStr);
const review = useEntityReview(entityIdStr);
const publications = useEntityPublications(entityIdStr);
const phenotypes = useEntityPhenotypes(entityIdStr);
const variation = useEntityVariation(entityIdStr);

// Linked gene fires when the entity record resolves with a hgnc_id.
const hgncIdRef = computed(() => {
  const e = entity.data.value as any;
  return e?.hgnc_id ?? null;
});
const linkedGene = useGeneRecord(hgncIdRef);
const linkedGeneSymbol = computed(() => (linkedGene.data.value as any)?.symbol?.[0] ?? '');
const linkedGeneName = computed(() => (linkedGene.data.value as any)?.name?.[0] ?? '');

// Publications client-side split.
const additionalRefs = computed(() =>
  ((publications.data.value as any[]) ?? []).filter(
    (p) => (p as any).publication_type === 'additional_references',
  ),
);
const geneReviews = computed(() =>
  ((publications.data.value as any[]) ?? []).filter(
    (p) => (p as any).publication_type === 'gene_review',
  ),
);

// Review empty when both synopsis and comment are blank.
const reviewIsEmpty = computed(() => {
  const r = review.data.value as any;
  if (!r) return true;
  const synopsisBlank = !r.synopsis || String(r.synopsis).trim() === '';
  const commentBlank = !r.comment || String(r.comment).trim() === '';
  return synopsisBlank && commentBlank;
});

// 404 redirect: only when the entity record returns null (canonical not-found).
watch(entity.data, (val) => {
  if (!entity.loading.value && val === null && entityIdStr.value) {
    router.push('/PageNotFound');
  }
});

const entityFields = [
  { key: 'symbol', label: 'Gene' },
  { key: 'disease_ontology_name', label: 'Disease' },
  { key: 'hpo_mode_of_inheritance_term_name', label: 'Inheritance' },
  { key: 'category', label: 'Status' },
];

useHead({
  title: computed(() => (entityIdStr.value ? `Entity: ${entityIdStr.value}` : 'Entity')),
  meta: [
    {
      name: 'description',
      content: computed(() =>
        entityIdStr.value
          ? `Entity ${entityIdStr.value} — gene-disease association in SysNDD.`
          : 'This Entity view shows specific information for a SysNDD entity.',
      ),
    },
  ],
});
</script>

}
</style>
```

## Task W3.4: EntityView spec — pin parallel hook fan-out

Locks: (1) all 5 sub-resource endpoints are hit on tick 0 (parallel, not sequential), (2) the publications result splits into `additionalRefs` and `geneReviews` from a single fetch, (3) the linked-gene block hydrates from the entity record's `hgnc_id`, (4) 404 redirect when the entity record is empty.

- [ ] **Step 1: Write the spec**

```typescript
// app/src/views/pages/__tests__/EntityView.spec.ts
import { describe, expect, it, beforeEach, afterEach, vi } from 'vitest';
import { setActivePinia, createPinia } from 'pinia';
import { http, HttpResponse } from 'msw';
import { mount, flushPromises } from '@vue/test-utils';
import { createMemoryHistory, createRouter } from 'vue-router';
import { server } from '@/test-utils/mocks/server';
import EntityView from '../EntityView.vue';

function makeRouter(path: string) {
  const router = createRouter({
    history: createMemoryHistory(),
    routes: [
      { path: '/Entities/:entity_id', name: 'Entity', component: EntityView },
      { path: '/PageNotFound', name: 'NotFound', component: { template: '<div>404</div>' } },
      { path: '/Genes/:symbol', name: 'Gene', component: { template: '<div/>' } },
    ],
  });
  router.push(path);
  return router;
}

describe('EntityView (v11.3 W3)', () => {
  beforeEach(() => setActivePinia(createPinia()));
  afterEach(() => server.resetHandlers());

  it('fires all 5 sub-resource calls in parallel on mount (no sequential await)', async () => {
    const calls: string[] = [];
    const start = Date.now();
    const tag = (path: string) => calls.push(`${Date.now() - start}:${path}`);

    server.use(
      http.get('*/api/entity/', () =>
        HttpResponse.json({ data: [{ entity_id: 304, symbol: 'MECP2', hgnc_id: 'HGNC:6990' }], links: [], meta: [{}] }),
      ),
      http.get('*/api/entity/304/status', () => {
        tag('status');
        return HttpResponse.json({ status: 'approved' });
      }),
      http.get('*/api/entity/304/review', () => {
        tag('review');
        return HttpResponse.json({ synopsis: '', comment: '' });
      }),
      http.get('*/api/entity/304/publications', () => {
        tag('pubs');
        return HttpResponse.json([]);
      }),
      http.get('*/api/entity/304/phenotypes', () => {
        tag('pheno');
        return HttpResponse.json([]);
      }),
      http.get('*/api/entity/304/variation', () => {
        tag('var');
        return HttpResponse.json([]);
      }),
      http.get('*/api/gene/HGNC:6990', () => HttpResponse.json([{ symbol: ['MECP2'] }])),
    );

    const router = makeRouter('/Entities/304');
    const w = mount(EntityView, { global: { plugins: [router] } });
    await flushPromises();

    // All five sub-resource handlers were hit.
    expect(calls.map((c) => c.split(':')[1]).sort()).toEqual(
      ['phenotypes'.slice(0, 5), 'pheno', 'pubs', 'review', 'status', 'var'].sort().filter((v, i, a) => a.indexOf(v) === i),
    );
    // Or simpler: each tag appeared at least once
    expect(calls.some((c) => c.endsWith('status'))).toBe(true);
    expect(calls.some((c) => c.endsWith('review'))).toBe(true);
    expect(calls.some((c) => c.endsWith('pubs'))).toBe(true);
    expect(calls.some((c) => c.endsWith('pheno'))).toBe(true);
    expect(calls.some((c) => c.endsWith('var'))).toBe(true);
    // No tag should be 100ms later than the earliest — they all started within the same tick.
    const offsets = calls.map((c) => Number(c.split(':')[0]));
    const spread = Math.max(...offsets) - Math.min(...offsets);
    expect(spread).toBeLessThan(50); // tight parallel spread
    w.unmount();
  });

  it('splits publications into additional_references and gene_review from one fetch', async () => {
    let pubsCalls = 0;
    server.use(
      http.get('*/api/entity/', () =>
        HttpResponse.json({ data: [{ entity_id: 304, hgnc_id: 'HGNC:6990' }], links: [], meta: [{}] }),
      ),
      http.get('*/api/entity/304/status', () => HttpResponse.json({})),
      http.get('*/api/entity/304/review', () => HttpResponse.json({ synopsis: '' })),
      http.get('*/api/entity/304/publications', () => {
        pubsCalls += 1;
        return HttpResponse.json([
          { publication_type: 'additional_references', pmid: '1' },
          { publication_type: 'additional_references', pmid: '2' },
          { publication_type: 'gene_review', pmid: '3' },
        ]);
      }),
      http.get('*/api/entity/304/phenotypes', () => HttpResponse.json([])),
      http.get('*/api/entity/304/variation', () => HttpResponse.json([])),
      http.get('*/api/gene/HGNC:6990', () => HttpResponse.json([{ symbol: ['MECP2'] }])),
    );

    const router = makeRouter('/Entities/304');
    const w = mount(EntityView, { global: { plugins: [router] } });
    await flushPromises();

    expect(pubsCalls).toBe(1);
    const html = w.html();
    expect(html).toContain('PMID:1');
    expect(html).toContain('PMID:2');
    expect(html).toContain('PMID:3');
    w.unmount();
  });

  it('redirects to /PageNotFound when the entity record is missing', async () => {
    server.use(
      http.get('*/api/entity/', () =>
        HttpResponse.json({ data: [], links: [], meta: [{}] }),
      ),
    );
    const router = makeRouter('/Entities/999999');
    const push = vi.spyOn(router, 'push');
    const w = mount(EntityView, { global: { plugins: [router] } });
    await flushPromises();
    expect(push).toHaveBeenCalledWith('/PageNotFound');
    w.unmount();
  });

  it('hydrates the linked-gene block from entity.hgnc_id', async () => {
    server.use(
      http.get('*/api/entity/', () =>
        HttpResponse.json({ data: [{ entity_id: 304, hgnc_id: 'HGNC:6990' }], links: [], meta: [{}] }),
      ),
      http.get('*/api/entity/304/status', () => HttpResponse.json({})),
      http.get('*/api/entity/304/review', () => HttpResponse.json({ synopsis: 'r' })),
      http.get('*/api/entity/304/publications', () => HttpResponse.json([])),
      http.get('*/api/entity/304/phenotypes', () => HttpResponse.json([])),
      http.get('*/api/entity/304/variation', () => HttpResponse.json([])),
      http.get('*/api/gene/HGNC:6990', () =>
        HttpResponse.json([{ symbol: ['MECP2'], name: ['methyl-CpG binding protein 2'] }]),
      ),
    );
    const router = makeRouter('/Entities/304');
    const w = mount(EntityView, { global: { plugins: [router] } });
    await flushPromises();
    expect(w.html()).toContain('MECP2');
    expect(w.html()).toContain('methyl-CpG binding protein 2');
    w.unmount();
  });
});
```
