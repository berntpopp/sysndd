### How to read an R/Plumber endpoint file

```bash
grep -nE '^#\* @(get|post|put|delete|patch) ' api/endpoints/<name>_endpoints.R
# Each match is a handler. The line below the handler-fn line is its R name.
# Read 30 lines of context to capture: query/body params, response shape, auth filter, serializer.
```

Common patterns:

- `#* @get path/<param>` — GET with a path parameter
- `#* @post path` + `#* @parser json` + a body argument — JSON-body POST
- `#* @serializer unboxedJSON` — response is a plain JSON value (not a 1-element array)
- `#* @preempt @authentication-required-Curator` — auth filter; the typed helper does NOT model auth (the apiClient interceptor injects the bearer token)
- `function(req, res, paramA, paramB = "default")` — the function signature names query/body params

### The typed-client contract template

Every filled module follows this template. Use it as the literal scaffold for each module:

```ts
// app/src/api/<name>.ts
//
// Mirrors api/endpoints/<name>_endpoints.R.
// Phase E.E2: filled during v11.1 Wave 1a (W3).

import type { AxiosRequestConfig } from 'axios';
import { apiClient } from './client';

// ---------------------------------------------------------------------------
// Types — mirror the R endpoint's response shape exactly
// ---------------------------------------------------------------------------

export interface FooSummary {
  /** numeric id from ndd_foo.foo_id */
  foo_id: number;
  /** display name */
  name: string;
  /** ISO 8601 from MySQL DATETIME */
  created_at: string;
}

// ---------------------------------------------------------------------------
// Helpers — one per @get/@post/@put/etc. handler
// ---------------------------------------------------------------------------

/**
 * GET /api/foo/<id>
 * Mirrors api/endpoints/foo_endpoints.R:NN (the line number of the handler)
 *
 * Returns the foo summary on 200. Throws AxiosError on non-2xx; callers
 * that want to map 404 to "no data" should catch and inspect via
 * isApiError + err.response?.status.
 */
export async function getFoo(
  fooId: number,
  config?: AxiosRequestConfig,
): Promise<FooSummary> {
  const path = `/api/foo/${encodeURIComponent(String(fooId))}`;
  return apiClient.get<FooSummary>(path, config);
}

/**
 * POST /api/foo
 * Mirrors api/endpoints/foo_endpoints.R:NN
 *
 * Creates a new foo and returns the created summary.
 */
export async function createFoo(
  body: { name: string },
  config?: AxiosRequestConfig,
): Promise<FooSummary> {
  return apiClient.post<FooSummary, { name: string }>('/api/foo', body, config);
}
```

Rules — verify each in the W3 self-review:

- Path params via `encodeURIComponent`
- Query params passed through `config.params`, never string-concatenated
- 404 NOT swallowed in the helper; documented in JSDoc as the caller's concern
- 1-element-array Plumber wrappers handled via `unwrapScalar` (from `client.ts`) when the endpoint omits `@serializer unboxedJSON`
- No retry/circuit-breaker logic
- Helpers throw on non-2xx (default `apiClient` behaviour)

### The Vitest + MSW spec template

```ts
// app/src/api/<name>.spec.ts
import { describe, test, expect, beforeAll, afterEach, afterAll } from 'vitest';
import { http, HttpResponse } from 'msw';
import { setupServer } from 'msw/node';

import { getFoo, createFoo, type FooSummary } from './foo';

const server = setupServer();

beforeAll(() => server.listen({ onUnhandledRequest: 'error' }));
afterEach(() => server.resetHandlers());
afterAll(() => server.close());

describe('api/foo: getFoo', () => {
  test('returns the response body on 200', async () => {
    const expected: FooSummary = { foo_id: 7, name: 'spec', created_at: '2026-01-01T00:00:00Z' };
    server.use(
      http.get('/api/foo/7', () => HttpResponse.json(expected)),
    );
    const result = await getFoo(7);
    expect(result).toEqual(expected);
  });

  test('throws AxiosError on 500', async () => {
    server.use(
      http.get('/api/foo/7', () => HttpResponse.json({ error: 'boom' }, { status: 500 })),
    );
    await expect(getFoo(7)).rejects.toThrow();
  });

  test('throws on 404 (caller decides whether to map to no-data)', async () => {
    server.use(
      http.get('/api/foo/999', () => HttpResponse.json({ error: 'not found' }, { status: 404 })),
    );
    await expect(getFoo(999)).rejects.toThrow();
  });
});

describe('api/foo: createFoo', () => {
  test('posts the body and returns the created entity', async () => {
    const created: FooSummary = { foo_id: 8, name: 'new', created_at: '2026-01-02T00:00:00Z' };
    server.use(
      http.post('/api/foo', async ({ request }) => {
        const body = await request.json();
        expect(body).toEqual({ name: 'new' });
        return HttpResponse.json(created);
      }),
    );
    const result = await createFoo({ name: 'new' });
    expect(result).toEqual(created);
  });
});
```

The existing repo already uses MSW (the `lint:wave1b-gate` and existing test setup confirm this). If MSW is not yet a dev dep, install it: `npm install --save-dev msw`.
