### Composable boundaries (target)

After this workstream, `Review.vue` orchestrates four composables:

| Composable | Owns | Public API (sketch) |
|---|---|---|
| `useReviewData.ts` | data fetching + cache state | `entries`, `loading`, `error`, `loadEntries()` |
| `useReviewFilters.ts` | filter state (text, role, status) | `filters`, `setFilter(...)`, `filteredEntries` |
| `useReviewModals.ts` | open/close state for approve / reject / detail dialogs | `isApproveOpen`, `currentTarget`, `openApprove(target)`, `close()` |
| `useReviewActions.ts` | approve/reject mutations + their loading state | `approve(target)`, `reject(target)`, `mutating` |

`Review.vue` itself only:

- Imports the four composables
- Wires them together in `<script setup>` or via the existing Options API integration
- Renders the table + filters + modals using template-only logic

The migration order matters: typed clients first, THEN extract composables. Otherwise composables will be migrated twice.

```ts
// app/src/views/review/composables/useReviewData.ts
import { ref, type Ref } from 'vue';
import { getReviewEntries, type ReviewEntry } from '@/api/review';

export interface UseReviewData {
  entries: Ref<ReviewEntry[]>;
  loading: Ref<boolean>;
  error: Ref<Error | null>;
  loadEntries: () => Promise<void>;
}

export function useReviewData(): UseReviewData {
  const entries = ref<ReviewEntry[]>([]);
  const loading = ref(false);
  const error = ref<Error | null>(null);

  async function loadEntries(): Promise<void> {
    loading.value = true;
    error.value = null;
    try {
      const result = await getReviewEntries();
      entries.value = result;
    } catch (err) {
      error.value = err instanceof Error ? err : new Error(String(err));
      throw err;
    } finally {
      loading.value = false;
    }
  }

  return { entries, loading, error, loadEntries };
}
```

```ts
// app/src/views/review/composables/useReviewFilters.ts
import { computed, reactive, type ComputedRef, type Ref } from 'vue';
import type { ReviewEntry } from '@/api/review';

type FilterKey = 'text' | 'status' | 'role';
type Filters = Record<FilterKey, string>;

export interface UseReviewFilters {
  filters: Filters;
  setFilter: (key: FilterKey, value: string) => void;
  filteredEntries: ComputedRef<ReviewEntry[]>;
}

export function useReviewFilters(source: Ref<ReviewEntry[]>): UseReviewFilters {
  const filters = reactive<Filters>({ text: '', status: '', role: '' });

  function setFilter(key: FilterKey, value: string): void {
    filters[key] = value;
  }

  const filteredEntries = computed(() => {
    return source.value.filter((e) => {
      if (filters.text && !(e.gene_symbol ?? '').toLowerCase().includes(filters.text.toLowerCase())) {
        return false;
      }
      if (filters.status && e.status !== filters.status) return false;
      if (filters.role && e.role_required !== filters.role) return false;
      return true;
    });
  });

  return { filters, setFilter, filteredEntries };
}
```

```ts
// app/src/views/review/composables/useReviewModals.ts
import { ref, type Ref } from 'vue';
import type { ReviewEntry } from '@/api/review';

export interface UseReviewModals {
  isApproveOpen: Ref<boolean>;
  isRejectOpen: Ref<boolean>;
  currentTarget: Ref<ReviewEntry | null>;
  openApprove: (target: ReviewEntry) => void;
  openReject: (target: ReviewEntry) => void;
  close: () => void;
}

export function useReviewModals(): UseReviewModals {
  const isApproveOpen = ref(false);
  const isRejectOpen = ref(false);
  const currentTarget = ref<ReviewEntry | null>(null);

  function openApprove(target: ReviewEntry): void {
    isRejectOpen.value = false;
    currentTarget.value = target;
    isApproveOpen.value = true;
  }
  function openReject(target: ReviewEntry): void {
    isApproveOpen.value = false;
    currentTarget.value = target;
    isRejectOpen.value = true;
  }
  function close(): void {
    isApproveOpen.value = false;
    isRejectOpen.value = false;
    currentTarget.value = null;
  }

  return { isApproveOpen, isRejectOpen, currentTarget, openApprove, openReject, close };
}
```

```ts
// app/src/views/review/composables/useReviewActions.ts
import { ref, type Ref } from 'vue';
import { approveReview, rejectReview } from '@/api/review';
import type { ReviewEntry } from '@/api/review';

export interface UseReviewActions {
  mutating: Ref<boolean>;
  approve: (target: ReviewEntry) => Promise<void>;
  reject: (target: ReviewEntry, reason?: string) => Promise<void>;
}

export function useReviewActions(): UseReviewActions {
  const mutating = ref(false);

  async function approve(target: ReviewEntry): Promise<void> {
    mutating.value = true;
    try {
      await approveReview(target.entity_id);
    } finally {
      mutating.value = false;
    }
  }

  async function reject(target: ReviewEntry, reason?: string): Promise<void> {
    mutating.value = true;
    try {
      await rejectReview(target.entity_id, reason);
    } finally {
      mutating.value = false;
    }
  }

  return { mutating, approve, reject };
}
```

```ts
// app/src/views/review/Review.vue (script excerpt)
import { onMounted } from 'vue';
import { useReviewData } from './composables/useReviewData';
import { useReviewFilters } from './composables/useReviewFilters';
import { useReviewModals } from './composables/useReviewModals';
import { useReviewActions } from './composables/useReviewActions';

export default defineComponent({
  setup() {
    const data = useReviewData();
    const filters = useReviewFilters(data.entries);
    const modals = useReviewModals();
    const actions = useReviewActions();

    onMounted(() => { void data.loadEntries(); });

    async function onApprove(target: ReviewEntry) {
      await actions.approve(target);
      modals.close();
      await data.loadEntries(); // refresh queue
    }
    async function onReject(target: ReviewEntry, reason?: string) {
      await actions.reject(target, reason);
      modals.close();
      await data.loadEntries();
    }

    return {
      ...data,
      ...filters,
      ...modals,
      mutating: actions.mutating,
      onApprove,
      onReject,
    };
  },
});
```

The `<template>` and `<style>` sections do not need to change structurally; they bind to the same names.
